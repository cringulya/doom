** Dired
#+begin_src emacs-lisp
(defun vertico-kill-buffer ()
  (interactive)
  (when-let ((buffer (get-buffer (vertico--candidate))))
    (kill-buffer buffer)
    (message "Killed buffer: %s" (buffer-name buffer))))

(map! :map vertico-map
      "C-d" #'vertico-kill-buffer)
#+end_src
